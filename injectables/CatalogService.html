import { Grouper } from './../../Helpers/Grouper';
import { DatabaseService } from './../database/database.service';
import { Injectable } from '@angular/core';
import { Catalog, Item, ItemModel } from '../../models/ui-model';
import { Settings } from '../../models/user';

@Injectable({
  providedIn: 'root',
})
export class CatalogService {
  constructor(private db: DatabaseService) {}

  getCatalog(catalogid: number, itemid: number, searchkey = '') {
    return this.getProductsGroupedByCategory(
      itemid,
      catalogid,
      searchkey,
      -1,
      -1
    );
  }

  getProductsCategoriesOnSystem(
    catalogid: number = 0,
    searchkey = '',
    page = 0,
    size = 0
  ) {
    return this.db.getProductsCategories(catalogid, searchkey, page, size);
  }

  getCatalogGeneric(
    catalogid: number,
    itemid: number,
    page: number,
    size: number,
    searchkey = '',
    barcodeid = ''
  ) {
    return this.getProductsGeneric(
      itemid,
      catalogid,
      searchkey,
      page,
      size,
      barcodeid,
      true
    );
  }

  getCatalogByBarcode(searchkey = '') {
    return this.getProductsGroupedByBarcode(searchkey);
  }


  async getProductsGroupedByCategory(
    product: any,
    categoryid: number,
    key: string,
    page: number,
    size: number
  ) {
    const res = await this.getProductsGeneric(
      product,
      categoryid,
      key,
      page,
      size,
      '',
      false
    );
    return this.finalCatalogResult(res);
  }

  async getProductsGeneric(
    product: any,
    categoryid: number,
    key: string,
    page: number,
    size: number,
    barcodeid = '',
    productOnly = false
  ) {
    const res = await this.db.getProductsAndCategories(
      product,
      categoryid,
      key,
      page,
      size,
      barcodeid,
      productOnly
    );

    if (res !== undefined && res.length > 0) {
      return res;
    } else {
      return this.defaultData();
    }
  }

  defaultData(): any[] {
    const data: any[] = [];
    return data;
  }

  async getSettings() {
    return this.db.getSettings();
  }

  async getProductsByBarcode(barcodeid: string) {
    const res = await this.db.getProductsByBarcode(barcodeid);
    return this.finalCatalogResult(res);
  }

  async finalCatalogResult(res: any[]) {
    if (res !== undefined && res !== null && res.length > 0) {
      const result = Grouper.groupByMultiple(res, function (item: any) {
        return [item.M_Product_Category_ID, item.C_Name];
      });

      // get settings
      const settings: Settings[] = await this.getSettings();
      const pricelistid = settings[0].DefaultPriceList;

      // get valid price list version for a price list
      const ver = await this.db.getPriceListVersion(pricelistid, 1);

      const values: any[] = [];
      for (const val of result) {
        const cat = new Catalog();
        cat.CatalogId = val[0].M_Product_Category_ID;
        cat.CatalogName = val[0].C_Name;
        cat.CatalogDescription = '';
        cat.CatalogImage = '';

        cat.CatalogItems = [];
        let isadd = true;
        for (const v of val) {
          isadd = true;
          const item = new Item();
          item.ItemId = v.M_Product_ID;
          item.ItemImage = v.ImageURL;
          item.ItemReference = v.Value;
          item.UPC = v.UPC;
          item.ItemModels = [];

          if (item.ItemId > 0) {
            if (this.db.issqlite) {
              // get product price based on price list version and product id
              const price = await this.db.getProductPrice(
                item.ItemId,
                ver[0].M_PriceList_Version_ID
              );

              // get product UOM conversions
              const uoms = await this.db.getProductUOM(item.ItemId, 0);
              if (uoms !== undefined && uoms !== null && uoms.length > 0) {
                for (const val2 of uoms) {
                  const model = new ItemModel();
                  model.C_UOM_ID = val2.C_UOM_ID;
                  model.ItemId = val2.M_Product_ID;
                  model.Model = val2.Name;

                  // check if standard price is set and use it. otherwise use the base price
                  // consider divideRate when it is available
                  if (
                    price !== undefined &&
                    price !== null &&
                    price.length > 0
                  ) {
                    if (
                      price[0].PriceStd === undefined ||
                      price[0].PriceStd === null ||
                      price[0].PriceStd === 0
                    ) {
                      model.Price = price[0].PriceList;
                    } else {
                      model.Price = price[0].PriceStd;
                    }
                  } else {
                    isadd = false;
                    model.Price = 0;
                  }

                  item.ItemModels.push(model);
                }
              }
            } else {
              const model = new ItemModel();
              model.C_UOM_ID = 100;
              model.ItemId = item.ItemId;
              model.Model = 'Each';
              model.Price = 1300;

              item.ItemModels.push(model);
            }
          }
          item.ItemName = v.P_Name;
          item.ItemDescription = '';

          if (
            item.ItemName !== undefined &&
            item.ItemName !== null &&
            item.ItemModels.length > 0 &&
            isadd
          ) {
            cat.CatalogItems.push(item);
          }
        }

        values.push(cat);
      }

      return values;
    } else {
      return this.defaultData();
    }
  }

  async getProductsGroupedByBarcode(barcodeid: string) {
    const res = await this.db.getProductsByBarcode(barcodeid);
    return this.finalCatalogResult(res);
  }
}
