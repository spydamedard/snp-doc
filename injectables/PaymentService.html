  async checkPaymentStatus(orderidForcheckingPaymentStatus: string) {
    if (orderidForcheckingPaymentStatus === null) {
      throw new Error('the order id cannot be null');
    }

    if (orderidForcheckingPaymentStatus === undefined) {
      throw new Error('please provide the order');
    }

    // we are saving the log for indicating that the system has started check the payment from the backend
    let createdForcheckingPaymentStatus = this.datepipe.transform(new Date(), 'yyyy-MM-dd HH:mm:ss');
    // upload order
    let logForcheckingPaymentStatus = new Log();
    logForcheckingPaymentStatus.L_created = createdForcheckingPaymentStatus;
    logForcheckingPaymentStatus.L_id = this.logServe.generateLogId();
    logForcheckingPaymentStatus.L_msg = ' Check payment status: Start check ' + createdForcheckingPaymentStatus + ' ...';
    logForcheckingPaymentStatus.L_key = 'order';
    logForcheckingPaymentStatus.L_data = orderidForcheckingPaymentStatus;
    this.logServe.newLog(logForcheckingPaymentStatus);
    const resForcheckingPaymentStatus = await this.orderService.checkIPaymentConfirmation(orderidForcheckingPaymentStatus);
    // we are saving the log immediatly after the backend's response
    createdForcheckingPaymentStatus = this.datepipe.transform(new Date(), 'yyyy-MM-dd HH:mm:ss');
    // upload order
    logForcheckingPaymentStatus = new Log();
    logForcheckingPaymentStatus.L_created = createdForcheckingPaymentStatus;
    logForcheckingPaymentStatus.L_id = this.logServe.generateLogId();
    logForcheckingPaymentStatus.L_msg =
      ' Check payment status: Successful . Backend response : ' +
      resForcheckingPaymentStatus +
      createdForcheckingPaymentStatus +
      ' ...';
    logForcheckingPaymentStatus.L_key = 'check_payment';
    logForcheckingPaymentStatus.L_data = resForcheckingPaymentStatus;
    await this.logServe.newLog(logForcheckingPaymentStatus);
    if (resForcheckingPaymentStatus !== null && resForcheckingPaymentStatus !== undefined) {
      return resForcheckingPaymentStatus;
    } else {
      throw new Error('cannot call backend process');
    }
  }

  private generatePaymentNo() {
    const yy = this.datepipe.transform(new Date(), 'yy');
    const mm = this.datepipe.transform(new Date(), 'M');
    const dd = this.datepipe.transform(new Date(), 'd');
    const h = this.datepipe.transform(new Date(), 'h');
    const m = this.datepipe.transform(new Date(), 'm');
    const s = this.datepipe.transform(new Date(), 's');
    let rand5 = '';
    let rand2 = '';
    const letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'; // include numbers if you want
    // tslint:disable-next-line: prefer-for-of
    for (let i = 0; i < 5; i++) {
      rand5 += letters.charAt(Math.floor(Math.random() * letters.length));
    }
    for (let i = 0; i < 2; i++) {
      rand2 += letters.charAt(Math.floor(Math.random() * letters.length));
    }
    const val = rand5 + yy + m + dd + rand2 + s + h;
    return val;
  }

      c_payTest.C_Order_ID = uidataTest.c_order_id;
      c_payTest.Description = '';
      c_payTest.IEXT_Receipt_Amnt = uidataTest.itext_reciept_amount;
      c_payTest.PayAmt = uidataTest.payAmount;
      c_payTest.Payment_Created_Date = new Date();
      c_payTest.Payment_Upload_Date = new Date();
      c_payTest.SalesRep_ID = user[0].AD_User_ID;
      if (uidataTest.flag === 'M') {
        // we are checking the phone number when using ipay
        if (uidataTest.phone === undefined) {
          throw new Error(
            'Mobile payment has been set, but the costumer phone number is undefined'
          );
        }

        if (uidataTest.phone === '') {
          throw new Error(
            'Mobile payment has been set, but the costumer phone number is empty'
          );
        }

        if (uidataTest.phone === null) {
          throw new Error(
            'Mobile payment has been set, but the costumer phone number is null'
          );
        }
        c_payTest.Phone = uidataTest.phone;
        c_payTest.TenderType = 'M';
      } else if (uidataTest.flag === 'P') {
        c_payTest.TenderType = defautldata.DefaultTenderType;
        c_payTest.Phone = '';
      } else {
        throw new Error('invalid tender type');
      }
      c_payTest.DocumentNO = '';
      if (uidataTest.payType === 'PO') {
        const res = await this.databaseServe.createPayment(c_payTest);
        if (res.rowsAffected === 1) {
          // upload order
          const uores = await this.orderService.uploadOrderData(
            uidataTest.c_order_id
          );
          if (uores !== undefined && uores !== null && uores !== '') {
            // upload payment
            await this.orderService.uploadPaymentData(
              uidataTest.c_order_id,
              uidataTest.flag
            );
          }
        }
      } else {
        const res = await this.databaseServe.createPayment(c_payTest);
        if (res.rowsAffected === 1) {
          await this.orderService.uploadPaymentData(
            uidataTest.c_order_id,
            uidataTest.flag
          );
        }
      }
    }
    return c_payTest;
  }

  public async returnLocalpayment(orderidForfetchingLocalPayment: string) {
    const paymentForfetchingLocalPayment = await this.databaseServe.getPayment(orderidForfetchingLocalPayment, 0);
    if (paymentForfetchingLocalPayment === undefined || paymentForfetchingLocalPayment === null || paymentForfetchingLocalPayment.length === 0) {
      throw new Error('no order payment info found');
    }
    return paymentForfetchingLocalPayment;
  }

    throw new Error(error);
  }
