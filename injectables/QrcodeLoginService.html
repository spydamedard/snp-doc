import { DefaultService } from '../default/default.service';
import { DatabaseService } from '../database/database.service';
import { Injectable } from '@angular/core';
import { UserInitialSetting } from '../../models/ui-model';

@Injectable({
  providedIn: 'root',
})

export class QrcodeLoginService {
  constructor(
    private db: DatabaseService
  ) {}

   /* ****************************************************************************
    setUserInitialSetting: this function set the user's initials setting
    @params
          uidata: the object that contains the the setting
    @return
          Promise <boolean>
  *******************************************************************************/

  async setUserInitialSetting(uidata: string): Promise<Boolean> {
    const userObject = JSON.parse(uidata);
    const user = new UserInitialSetting();

    let result_rs = false;

    if (
      userObject.CLIENTID === undefined ||
      userObject.ORGID === undefined ||
      userObject.ROLEID === undefined ||
      userObject.WAREHOUSEID === undefined ||
      userObject.STAGE === undefined ||
      userObject.BACKENDURL === undefined
    ) {
      throw 'please be sure that all variables are set';
    }

    user.CLIENTID = userObject.CLIENTID;

    user.ORGID = userObject.ORGID;

    user.ROLEID = userObject.ROLEID;

    user.WAREHOUSEID = userObject.WAREHOUSEID;

    user.STAGE = userObject.STAGE;

    user.BACKENDURL = userObject.BACKENDURL;

    const result_persis = await this.db.AddDefaultUserSetting(user);

    if (result_persis.rowsAffected === 1) {
      await this.db.AddDefaultUserSettingLogin(user);
      result_rs = true;
    }

    return result_rs;
  }

  /*
  User initial setting fething
  */
  async getInitialUserSettings() {
    const Usettings = await this.db.getUserInitialSetting();
    return Usettings[0];
  }

   /*
  User initial login setting fething
  */
  async getInitialUserSettingsLogin() {
    const Usettings = await this.db.getUserInitialSettingLogin();
    return Usettings[0];
  }
  private handleError(error: any) {
    return error;
  }
}
