import { DefaultService } from './../default/default.service';
import { DatabaseService } from './../database/database.service';
import { Urls } from './../../Helpers/Urls';
import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { ADLoginRequest } from '../../models/ADLoginRequest';
import { ModelCRUD } from '../../models/ModelCRUD';
import { ModelCRUDRequest, MyData } from '../../models/ModelCRUDRequest';
import { catchError } from 'rxjs/operators';
import { firstValueFrom } from 'rxjs';
import { ADUser, CurrentQRcodeSetting } from '../../models/user';
import { QrcodeLoginService } from '../qrcodelogin/qrcodelogservice';
import { Checker } from '../../models/ui-model';
import { DataService } from '../data/data.service';
import { LogService } from '../Logs/log.service';
import { DatePipe } from '@angular/common';
import { Log } from '../../models/log';

@Injectable({
  providedIn: 'root',
})
export class LoginService {
  constructor(
    private http: HttpClient,
    private db: DatabaseService,
    private qrcodelogin: QrcodeLoginService,
    private dservice: DataService,
    private dfaultS: DefaultService,
    private datepipe: DatePipe,
    private logServe: LogService
  ) {}

  private authenticateUser(lrequest: ADLoginRequest) {
    // model crud data
    const cdataForAuthentification = new ModelCRUD();
    cdataForAuthentification.serviceType = 'QueryUserData';
    cdataForAuthentification.TableName = 'AD_User';
    cdataForAuthentification.Action = 'Read';

    // request data
    const requestdataForAuthentification = new ModelCRUDRequest();
    requestdataForAuthentification.ModelCRUD = cdataForAuthentification;
    requestdataForAuthentification.ADLoginRequest = lrequest;

    // api request data
    const dataForAuthentification = new MyData();
    dataForAuthentification.ModelCRUDRequest = requestdataForAuthentification;

    // make api request to login and get user data
    return this.http
      .post<any>(Urls.apiUrl, dataForAuthentification)
      .pipe(catchError(this.handleError));
  }

  async updateBackendUrl(): Promise<boolean> {
    if (this.db.issqlite) {
      const user_settingForUpdatingBackendUrl = await this.qrcodelogin.getInitialUserSettings();
      if (user_settingForUpdatingBackendUrl !== undefined && user_settingForUpdatingBackendUrl !== null) {
        Urls.apiUrl =
          user_settingForUpdatingBackendUrl.BACKENDURL +
          '/' +
          'ADInterface/services/rest/model_adservice/query_data';
        Urls.apiUrl2 =
          user_settingForUpdatingBackendUrl.BACKENDURL +
          '/' +
          'ADInterface/services/rest/model_adservice';
        return true;
      } else {
        return false;
      }
    } else {
      return false;
    }
  }

  // login
  async login(email: string, password: string, language: string): Promise<any> {
    if (email === undefined || email === null || email.trim() === '') {
      throw 'provide username';
    } else if (
      password === undefined ||
      password === null ||
      password.trim() === ''
    ) {
      throw 'provide password';
    } else if (
      language === undefined ||
      language === null ||
      language.trim() === ''
    ) {
      throw 'provide language';
    } else {
      //initials user setting
      const user_settingForMakingAuthentification = await this.qrcodelogin.getInitialUserSettingsLogin();
      // we are creatiing the log for showing that the authentification has started - at this level the system is fetching the user default setting
      let createdForMakingAuthentification = this.datepipe.transform(new Date(), 'yyyy-MM-dd HH:mm:ss');
      let logForMakingAuthentification = new Log();
      logForMakingAuthentification.L_created = createdForMakingAuthentification;
      logForMakingAuthentification.L_id = this.logServe.generateLogId();
      logForMakingAuthentification.L_msg ='   Login: start fetching initial user settings ' + createdForMakingAuthentification + ' ...';
      logForMakingAuthentification.L_key = 'user_initial_settings';
      logForMakingAuthentification.L_data = JSON.stringify(user_settingForMakingAuthentification);
      await this.logServe.newLog(logForMakingAuthentification);
      // user data
      if (user_settingForMakingAuthentification !== undefined && user_settingForMakingAuthentification !== null) {
        // set backend url
        await this.updateBackendUrl();
        const lrequestForMakingAuthentification = new ADLoginRequest();
        lrequestForMakingAuthentification.user = email;
        lrequestForMakingAuthentification.pass = password;
        lrequestForMakingAuthentification.lang = language;
        lrequestForMakingAuthentification.ClientID = user_settingForMakingAuthentification.CLIENTID;
        lrequestForMakingAuthentification.RoleID = user_settingForMakingAuthentification.ROLEID;
        lrequestForMakingAuthentification.OrgID = user_settingForMakingAuthentification.ORGID;
        lrequestForMakingAuthentification.WarehouseID = user_settingForMakingAuthentification.WAREHOUSEID;
        lrequestForMakingAuthentification.stage = user_settingForMakingAuthentification.STAGE;
        // we are saving the log for indicating that at this level the system has started call the backend authentification function
        createdForMakingAuthentification = this.datepipe.transform(new Date(), 'yyyy-MM-dd HH:mm:ss');
        logForMakingAuthentification = new Log();
        logForMakingAuthentification.L_created = createdForMakingAuthentification;
        logForMakingAuthentification.L_id = this.logServe.generateLogId();
        logForMakingAuthentification.L_msg =
          '   Login: start authenticate the user from the backend ' +
          createdForMakingAuthentification +
          ' ...';
        logForMakingAuthentification.L_key = 'login_request';
        logForMakingAuthentification.L_data = JSON.stringify(lrequestForMakingAuthentification);
        await this.logServe.newLog(logForMakingAuthentification);
        const dataForMakingAuthentification = await firstValueFrom(
          this.authenticateUser(lrequestForMakingAuthentification)
        ).catch((err) => this.handleError(err));

        // we are saving the log immediatly after the backend's response

        createdForMakingAuthentification = this.datepipe.transform(new Date(), 'yyyy-MM-dd HH:mm:ss');
        logForMakingAuthentification = new Log();
        logForMakingAuthentification.L_created = createdForMakingAuthentification;
        logForMakingAuthentification.L_id = this.logServe.generateLogId();
        logForMakingAuthentification.L_msg =
          '   Login: user  authentification from the backend ' +
          createdForMakingAuthentification;
        logForMakingAuthentification.L_key = 'login_request';
        logForMakingAuthentification.L_data = JSON.stringify(dataForMakingAuthentification);

        await this.logServe.newLog(logForMakingAuthentification);
        if (
          dataForMakingAuthentification !== undefined &&
          dataForMakingAuthentification !== null &&
          dataForMakingAuthentification.WindowTabData !== undefined &&
          dataForMakingAuthentification.WindowTabData !== null &&
          dataForMakingAuthentification.WindowTabData.DataSet !== undefined &&
          dataForMakingAuthentification.WindowTabData.DataSet !== null &&
          dataForMakingAuthentification.WindowTabData.DataSet.DataRow !== undefined &&
          dataForMakingAuthentification.WindowTabData.DataSet.DataRow !== null &&
          dataForMakingAuthentification.WindowTabData.DataSet.DataRow.field !== undefined &&
          dataForMakingAuthentification.WindowTabData.DataSet.DataRow.field !== null &&
          dataForMakingAuthentification.WindowTabData.DataSet.DataRow.field.length > 0
        ) {
          const userForMakingAuthentification = new ADUser();

          for (const valForMakingAuthentification of dataForMakingAuthentification.WindowTabData.DataSet.DataRow.field) {
            if (valForMakingAuthentification['@column'] === 'AD_User_ID') {
              userForMakingAuthentification.AD_User_ID = valForMakingAuthentification.val;
            }
            if (valForMakingAuthentification['@column'] === 'Name') {
              userForMakingAuthentification.Name = valForMakingAuthentification.val;
            }
            if (valForMakingAuthentification['@column'] === 'C_BPartner_ID') {
              userForMakingAuthentification.C_BPartner_ID = valForMakingAuthentification.val;
            }
          }
          // delete request data locally
          await this.db.deleteRequestData();
          // save request data locally
          await this.db.saveRequestData(lrequestForMakingAuthentification);
          // delete user data locally
          await this.db.deleteUser();
          // save user data locally
          await this.db.saveUser(userForMakingAuthentification);
          // delete current checker setting
          await this.db.deleteOrgWarehouseChecker(123);
          // create new default checker settings
          const org_and_warehouse_checker = new Checker();
          org_and_warehouse_checker.id = 123;
          org_and_warehouse_checker.org = 1;
          org_and_warehouse_checker.warehouse = 1;
          await this.db.CreateOrg_WareHouse_Checker(org_and_warehouse_checker);
        }
        return Promise.resolve(dataForMakingAuthentification);
      } else {
        throw new Error('default data not available');
      }
    }
  }

  /******************************************************************************************                                                          *
   * fetchQRcodeInfos: this function is called by the UI for fetching the QRcode information *
   *        @params:                                                                         *
   *                 empty                                                                   *
   *      @returned: <Array>                                                                 *
   ****************************************************************************************/
  async fetchQRcodeInfos(): Promise<CurrentQRcodeSetting> {
    //qrcodesetting instantiation
    let qrcodesettingVariableForCheckingQrInfoAvailability = new CurrentQRcodeSetting();
    // current organization and warehouse names
    const OrgWarHVariableForCheckingQrInfoAvailability = await this.dservice
      .getCurrentOrgAndWarehouse()
      .catch((error) => {
        this.handleError(error);
      });
    // default settings
    const dsettingVariableForCheckingQrInfoAvailability = await this.dfaultS.getDefaultSettings();
    // qrcode infos
    const dqrVariableForCheckingQrInfoAvailability = await this.qrcodelogin.getInitialUserSettings();
    // request data
    const lrequestVariableForCheckingQrInfoAvailability = await this.db.getRequestData();
    if (
      OrgWarHVariableForCheckingQrInfoAvailability !== null &&
      OrgWarHVariableForCheckingQrInfoAvailability !== undefined &&
      dsettingVariableForCheckingQrInfoAvailability !== null &&
      dsettingVariableForCheckingQrInfoAvailability !== undefined &&
      dqrVariableForCheckingQrInfoAvailability !== null &&
      dqrVariableForCheckingQrInfoAvailability !== undefined &&
      lrequestVariableForCheckingQrInfoAvailability !== null &&
      lrequestVariableForCheckingQrInfoAvailability !== undefined
    ) {
      if (OrgWarHVariableForCheckingQrInfoAvailability.org !== undefined && OrgWarHVariableForCheckingQrInfoAvailability.warehouse !== undefined) {
        qrcodesettingVariableForCheckingQrInfoAvailability.orgid = dqrVariableForCheckingQrInfoAvailability.ORGID;
        qrcodesettingVariableForCheckingQrInfoAvailability.warehouseid = dqrVariableForCheckingQrInfoAvailability.WAREHOUSEID;
        qrcodesettingVariableForCheckingQrInfoAvailability.org = OrgWarHVariableForCheckingQrInfoAvailability.org;
        qrcodesettingVariableForCheckingQrInfoAvailability.warehouse = OrgWarHVariableForCheckingQrInfoAvailability.warehouse;
        qrcodesettingVariableForCheckingQrInfoAvailability.client_name = dsettingVariableForCheckingQrInfoAvailability.Name;
        qrcodesettingVariableForCheckingQrInfoAvailability.clientid = dqrVariableForCheckingQrInfoAvailability.CLIENTID;
        qrcodesettingVariableForCheckingQrInfoAvailability.weburl = dsettingVariableForCheckingQrInfoAvailability.URL;
        qrcodesettingVariableForCheckingQrInfoAvailability.user_name = lrequestVariableForCheckingQrInfoAvailability[0].user;
      }
    }

    return qrcodesettingVariableForCheckingQrInfoAvailability;
  }
