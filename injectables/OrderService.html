  async updateOrderLineValues(
    orderid: string,
    lineId: string,
    qty: number,
    newprice: number,
    alsoCustomerPrice: boolean,
    temptotal: number
  ) {
    // update line quantity and net amount
    const res = await this.db.updateOrderLineItem(lineId, qty, newprice);

    if (res.rowsAffected < 1) {
      throw new Error('could not update order line item quantity and price');
    }

    // update actual price
    await this.db.updateOrderLine(lineId, 'actualprice', newprice.toString());

    // update customer's price
    if (alsoCustomerPrice) {
      await this.db.updateOrderLine(
        lineId,
        'CustomerPrice',
        newprice.toString()
      );
    }

    // update order grand total with new total based on line net amount
    await this.db.updateOrderWhnOrderlineChange(orderid);
    // return true;
    return this.getOLine(orderid, lineId);
  }

        productid_fr_ch,

        uomid_fr_ch

        uom_fr_ch === undefined ||
        uom_fr_ch === null ||

  async updateOrder(
    ord_id: string,
    status: string,
    grandtotal: number,
    amountentered = 0,
    isuploaded = 0,
    ispaid = 0,
    ispaiduploaded = 0
  ) {
    if (this.db.issqlite) {
      if (ord_id === undefined || ord_id === null || ord_id.trim() === '') {
        throw new Error('invalid order id');
      }

      if (
        (status === undefined || status === null || status.trim() === '') &&
        (grandtotal === undefined || grandtotal === null || grandtotal === 0) &&
        (amountentered === undefined ||
          amountentered === null ||
          amountentered === 0)
      ) {
        throw new Error('invalid update info');
      }

      let responseForUpdatingTheOrder = false;

      if (status !== undefined && status !== null && status.trim() !== '') {
        // update status
        const resForUpdatingTheOrderStatus = await this.db.updateOrder(
          ord_id,
          'status',
          status
        );
        if (resForUpdatingTheOrderStatus.rowsAffected === 1) {
          responseForUpdatingTheOrder = true;
        } else {
          throw new Error('could not update status');
        }
      }

      if (grandtotal !== undefined && grandtotal !== null && grandtotal !== 0) {
        // update grandtotal
        const resForUpdatingTheOrderGrandTotal = await this.db.updateOrder(
          ord_id,
          'grandtotal',
          grandtotal.toString()
        );
        if (resForUpdatingTheOrderGrandTotal.rowsAffected === 1) {
          responseForUpdatingTheOrder = true;
        } else {
          throw new Error('could not update grand total');
        }
      }

      if (
        amountentered !== undefined &&
        amountentered !== null &&
        amountentered !== 0
      ) {
        // update amountentered
        const resForUpdatingTheOrderAmountEntered = await this.db.updateOrder(
          ord_id,
          'amountentered',
          amountentered.toString()
        );
        if (resForUpdatingTheOrderAmountEntered.rowsAffected === 1) {
          responseForUpdatingTheOrder = true;
        } else {
          throw new Error('could not update amount entered');
        }
      }

      if (isuploaded > 0) {
        // update upload date
        const resForUpdatingTheOrderUploadDate = await this.db.updateOrder(
          ord_id,
          'uploaddate',
          ''
        );
        if (resForUpdatingTheOrderUploadDate.rowsAffected === 1) {
          responseForUpdatingTheOrder = true;
        } else {
          throw new Error('could not update upload date');
        }
      }

      if (ispaid > 0) {
        // update is paid
        const resForUpdatingTheOrderPaidStatus = await this.db.updateOrder(
          ord_id,
          'paid',
          ''
        );
        if (resForUpdatingTheOrderPaidStatus.rowsAffected === 1) {
          responseForUpdatingTheOrder = true;
        } else {
          throw new Error('could not update is paid');
        }
      }

          responseForUpdatingTheOrder = true;

      return Promise.resolve(false);
    }

  EntegerAndFloatPart(element: string) {
    let entegerside = '';
    let floatside = '';
    const leng = element.length;
    let k = 0;
    let cptCommaSection = 0;
    let checkComma = false;
    while (k <= leng) {
      if (
        element.charAt(k) !== undefined &&
        element.charAt(k) !== ',' &&
        !checkComma
      ) {
        entegerside += element.charAt(k);
      } else if (element.charAt(k) !== undefined && element.charAt(k) === ',') {
        checkComma = true;
      }

      if (checkComma) {
        cptCommaSection += 1;
      }

      if (cptCommaSection > 1) {
        floatside += element.charAt(k);
      }
      k += 1;
    }

    const conv = new ConverT();
    conv.integerSide = entegerside;
    conv.floatSide = floatside;

    return conv;
  }

  async increaseOrderLineItemQuantityForTest(
    orderid: string,
    lineid: string,
    productid: number,
    quant: number
  ) {
    this.checkOrderParameters(orderid, lineid, 1, productid);
    // the surfix 'IncrFT = increase order line item for test' has been added to each variable for specifying this context
    // get order
    const orderIncrFT = await this.db.getOrder(orderid, '', '', '', '');
    if (
      orderIncrFT === undefined ||
      orderIncrFT === null ||
      orderIncrFT.length === 0
    ) {
      throw new Error('order not found');
    }
    // get order line item
    const lineIncrFT = await this.fetchOrderLineInfo(
      orderid,
      lineid,
      productid
    );
    const gtotalIncrFT = orderIncrFT[0].GrandTotal;
    const netamtIncrFT = lineIncrFT[0].LineNetAmt;
    const temptotalIncrFT = gtotalIncrFT - netamtIncrFT;
    const qtyIncrFT = quant;
    const uomIncrFT = await this.db.getProductUOM(
      productid,
      lineIncrFT[0].C_UOM_ID,
      0
    );
    let priceIncrFT = 0;
    if (uomIncrFT[0].DivideRate === 0) {
      priceIncrFT = lineIncrFT[0].PriceActual;
    } else {
      priceIncrFT = lineIncrFT[0].PriceActual * uomIncrFT[0].DivideRate;
    }
    const newpriceIncrFT = priceIncrFT * qtyIncrFT;
    // update line quantity
    const resIncrFT = await this.db.updateOrderLine(
      lineIncrFT[0].C_OrderLine_ID,
      'quantity',
      qtyIncrFT.toString()
    );
    if (resIncrFT.rowsAffected !== 1) {
      throw new Error('could not update order line item quantity');
    }
    // update line netamount
    const res2IncrFT = await this.db.updateOrderLine(
      lineIncrFT[0].C_OrderLine_ID,
      'netamount',
      newpriceIncrFT.toString()
    );
    if (res2IncrFT.rowsAffected !== 1) {
      throw new Error('could not update order line item net amount');
    }
    // add order line total price to order grand total
    const totalIncrFT = temptotalIncrFT + newpriceIncrFT;
    let newvalIncrFT = totalIncrFT;
    newvalIncrFT = totalIncrFT === 0 ? -1 : totalIncrFT;
    // update order grand total with new total
    await this.updateOrder(orderid, '', newvalIncrFT, 0);
    return true;
  }
