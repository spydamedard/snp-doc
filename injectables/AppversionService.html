import { DatabaseService } from '../database/database.service';
import { DataService } from './../data/data.service';
import {
  AppVersion,
  AppVersionUIResponse,
  LastTimeFromTheBackend,
} from './../../models/ui-model';
import { Injectable } from '@angular/core';
import { ModelCRUDPayCompare } from '../../models/ModelCRUD';
import { firstValueFrom } from 'rxjs';
import { ModelCRUDRequest, MyData } from '../../models/ModelCRUDRequest';
import { DataRow, Field } from '../../models/ResponseData';
import { InitDataService } from '../web-services/init-data-service/init-data.service';
import { DatePipe } from '@angular/common';
/**
  Provider for super difficult math operations
*/

@Injectable({
  providedIn: 'root',
})
export class AppversionService {
  constructor(
    private databaseServe: DatabaseService,
    private datepipe: DatePipe,
    private initdataService: InitDataService,
    private DataServe: DataService
  ) {}

/**
* get app version from the backend
*
* @getAppversionFromtheBackend
* Simply call it with the version that you want to check:
* saveAppVersionStatus(1.2-34)
* @param {string} version First number
* @returns The the current version information
*/

  public async saveAppVersionStatus(
    appversion: string
  ): Promise<AppVersionUIResponse> {
    // model crud data
    const cdata = new ModelCRUDPayCompare();
    cdata.serviceType = 'QueryMobileVersionByVersion';
    cdata.TableName = 'iNO_MobileVersion';
    cdata.Action = 'Read';
    const row = new DataRow();
    let fields: Field[] = [];
    const onefield = new Field();
    onefield['@column'] = 'Version';
    onefield.val = appversion;
    fields.push(onefield);
    row.field = fields;
    cdata.DataRow = row;
    const lrequest = await this.databaseServe.getRequestData();
    // request data
    const requestdata = new ModelCRUDRequest();
    requestdata.ModelCRUD = cdata;
    requestdata.ADLoginRequest = lrequest[0];
    // api request data
    const data = new MyData();
    data.ModelCRUDRequest = requestdata;
    const UIRESP = new AppVersionUIResponse();
    const appversionData = await firstValueFrom(
      this.initdataService.requestData(data)
    ).catch((err) => this.handleError(err));
    let backendResponse: any = null;
    backendResponse = this.getAppversionAfterBackendResponse(appversionData);
    this.checkerSaveAppVersion(backendResponse);
    const finalVersionToSave=this.getAppversionObjectBasedOnBackendData(backendResponse);
    await this.databaseServe.CreateNewAppVersion(finalVersionToSave);
    const lastTime = new LastTimeFromTheBackend();
    lastTime.lasttimeBackendCheckAppVersion = this.datepipe.transform(
      new Date(),
      'yyyy-MM-dd HH:mm:ss'
    );
    await this.databaseServe.CreateNewLastTimeBackendAppVersionCheck(lastTime);
    UIRESP.Valid = finalVersionToSave.Approved;
    UIRESP.Comment = finalVersionToSave.Comment;
    return UIRESP;
  }

// set appversion object to save
getAppversionObjectBasedOnBackendData(backendResult:any)
{
  let version = new AppVersion();
  version.Comment = '';
  version.App_Version = '';
  version.Approved = '';
  for (const val of backendResult) {
    const previewApproved = version.Approved;
    const previewVersion = version.App_Version;
    const previewComment = version.Comment;
    version.Approved =
      val['@column'] === 'IsApproved' ? val.val : previewApproved;
    version.App_Version =
      val['@column'] === 'Version' ? val.val : previewVersion;
    if (val['@column'] === 'Comments') {
      version.Comment = !JSON.stringify(val.val).includes('@nil')
        ? val.val
        : previewComment;
    }
  }
  return version;
}
// this function is used for get the appversion from the backend's response data
  getAppversionAfterBackendResponse(appversionData:any)
  {
    let finalDataAfterCheck = null;
    if (
      appversionData !== undefined &&
      appversionData !== null &&
      appversionData.WindowTabData !== undefined &&
      appversionData.WindowTabData !== null &&
      appversionData.WindowTabData.DataSet !== undefined &&
      appversionData.WindowTabData.DataSet !== null &&
      appversionData.WindowTabData.DataSet.DataRow !== undefined &&
      appversionData.WindowTabData.DataSet.DataRow !== null
    ) {
      if (appversionData.WindowTabData.DataSet.DataRow.length > 0) {
        finalDataAfterCheck = appversionData.WindowTabData.DataSet.DataRow;
      } else {
        if (
          appversionData.WindowTabData.DataSet.DataRow.field !== undefined &&
          appversionData.WindowTabData.DataSet.DataRow.field !== null &&
          appversionData.WindowTabData.DataSet.DataRow.field.length > 0
        ) {
          finalDataAfterCheck = appversionData.WindowTabData.DataSet.DataRow.field;
        }
      }

    }
    return finalDataAfterCheck;
  }

  checkerSaveAppVersion(backstatus: string) {
    if (backstatus === null) {
      throw new Error('The version does not exists in the system');
    }
  }

  delay(ms: number) {
    return new Promise((resolve) => setTimeout(resolve, ms));
  }
  // check app version status

  async checkAppVersionStatus(appversion: string) {
    await this.checkerOfAppVersionParams(appversion);
    let UIRESP = new AppVersionUIResponse();
    // by default, we are supposing that all the versions are allowed
    UIRESP.Valid = 'Y';
    UIRESP.Comment = 'All versions are allowed';
    // we are checking the version from the backend
    await this.saveAppVersionStatus(appversion);
    await this.DataServe.getAndSaveSettingsData();
    // we are checking the app version strategy
    const versionStrategyFinalFetch =
      await this.databaseServe.getAppVersionStrategy();
    // we can continue if the time is not yet reached 8 hours after the last check from the backend
    if (
      versionStrategyFinalFetch !== null &&
      versionStrategyFinalFetch[0].IsActive === 'Y'
    ) {
      const versionFinal = await this.databaseServe
        .getAppVersion()
        .catch((error) => {
          this.handleError(error);
        });
      if (
        versionFinal !== undefined &&
        versionFinal !== null &&
        versionFinal.length > 0
      ) {

        UIRESP.Valid = versionFinal[0].Approved;
        UIRESP.Comment = versionFinal[0].Comment;
      }
    }
    return UIRESP;
  }

  async checkerOfAppVersionParams(appversion: string) {
    if (!this.databaseServe.issqlite) {
      throw new Error('Could not connect to sqlite');
    }
    if (appversion === undefined) {
      throw new Error('The app version is undefined');
    }
    if (appversion === '') {
      throw new Error('The app version is empty');
    }
    if (appversion === null) {
      throw new Error('The app version is null');
    }
  }

  async clearAPPversion() {
    await this.databaseServe.DeleteAppVersionAndAppStrategy();
    await this.databaseServe.DeleteAppVersion();
    await this.databaseServe.DeleteAppVersionLastBackendDownload();
  }

  async saveLastTimeForTest() {
    const lastTime = new LastTimeFromTheBackend();
    const hours = new Date().getHours() / 1000;
    const vdate = new Date().setHours(hours);
    lastTime.lasttimeBackendCheckAppVersion = this.datepipe.transform(
      vdate,
      'yyyy-MM-dd HH:mm:ss'
    );
    await this.databaseServe.CreateNewLastTimeBackendAppVersionCheck(lastTime);
  }

  private handleError(error: any) {
    return error;
  }
}
