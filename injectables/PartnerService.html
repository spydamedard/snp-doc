import {
  BusinessBartner,
  City,
  Group,
  Categorie,
} from './../../models/ui-model';
import { DatabaseService } from './../database/database.service';
import { Injectable } from '@angular/core';
import { CBPartner } from '../../models/org-data';

@Injectable({
  providedIn: 'root',
})
export class PartnerService {
  constructor(private db: DatabaseService) {}

  checkPartnerCategoryParams(data: number) {
    if (data === undefined) {
      throw new Error('please provide a  business partner categorie');
    }
    if (data < 0) {
      throw new Error('please provide a valid business partner categorie');
    }
  }

  checkPartnerCityParams(data: number) {
    if (data === undefined) {
      throw new Error('please provide a business partner city');
    }
    if (data < 0) {
      throw new Error('please provide a valid business partner city');
    }
  }

  checkPartnerIsCompanyParams(data: string) {
    if (data === undefined) {
      throw new Error(
        'please check if this business partner it"s a company or not'
      );
    }
    if (data === '') {
      throw new Error('the company checker cannot be empty');
    }
    if (data === null) {
      throw new Error('the company checker cannot be null');
    }
  }

  checkPartnerCompanyContribuableParams(data: string) {
    if (data === undefined) {
      throw new Error('please provide a business partner contribuable');
    }
    if (data === '') {
      throw new Error('business partner contribuable cannot be empty');
    }
    if (data === null) {
      throw new Error('business partner contribuable cannot be null');
    }
  }

  checkPartnerCompanyCommerceRegisterParams(data: string) {
    if (data === undefined) {
      throw new Error('please provide a business partner registre de commerce');
    }
    if (data === '') {
      throw new Error('business partner registre de commerce cannot be empty');
    }
  }

  checkPartnerCompanyCustomerNameParams(data: string) {
    if (data === undefined) {
      throw new Error('please provide a business partner name');
    }
    if (data === '') {
      throw new Error('business partner name cannot be empty');
    }
    if (data === null) {
      throw new Error('business partner name cannot be null');
    }
  }

  checkPartnerCompanyQuaterParams(data: string) {
    if (data === undefined) {
      throw new Error('please provide a business partner quater');
    }
    if (data === '') {
      throw new Error('business partner quater cannot be empty');
    }
    if (data === null) {
      throw new Error('business partner quater cannot be null');
    }
  }

  checkPartnerContacts(phone: string, mail: string, bp_user_name: string) {
    if (mail === undefined) {
      throw new Error('please provide a business partner user email');
    }
    if (bp_user_name === undefined) {
      throw new Error('please provide a business partner user name');
    }
    if (phone === undefined) {
      throw new Error('please provide a business partner user phone');
    }
  }

  async createBusinessPartner(
    part: BusinessBartner,
    orderid: string
  ): Promise<number> {
    this.checkPartnerCategoryParams(part.BP_Categorie);
    this.checkPartnerCityParams(part.BP_City);
    this.checkPartnerIsCompanyParams(part.BP_IsCompany);
    if (part.BP_IsCompany === 'Y') {
      this.checkPartnerCompanyContribuableParams(part.BP_IEXT_Contribuable);
      this.checkPartnerCompanyCommerceRegisterParams(
        part.BP_IEXT_Registre_Commerce
      );
      this.checkPartnerCompanyCustomerNameParams(part.BP_Name);
      this.checkPartnerCompanyQuaterParams(part.BP_Quarter);
    }
    this.checkPartnerContacts(
      part.BP_User_phone,
      part.BP_User_email,
      part.BP_User_name
    );
    return 0;
  }
  async UpdateOrderBusinessPartner(
    partnerid: number,
    orderid: string
  ): Promise<boolean> {
    return false;
  }

  async getBusinnessPartners(): Promise<CBPartner[]> {
    if (this.db.issqlite) {
      const partners = await this.db.getBPartner();
      if (partners !== undefined && partners !== null && partners.length > 0) {
        return Promise.resolve(partners);
      }
    }
    return Promise.resolve(this.defaultData());
  }

  async getBusinnessPartnersCity(): Promise<City[]> {
    if (this.db.issqlite) {
      let partnersLoc = await this.db.getPartnerCity();
      if (
        partnersLoc !== undefined &&
        partnersLoc !== null &&
        partnersLoc.length > 0
      ) {
        return Promise.resolve(partnersLoc);
      }
    }
    return Promise.resolve(this.defaultData());
  }

  async getBusinnessPartnerGroup(): Promise<Group[]> {
    if (this.db.issqlite) {
      const partnersgroup = await this.db.getPartnerGroup();
      if (
        partnersgroup !== undefined &&
        partnersgroup !== null &&
        partnersgroup.length > 0
      ) {
        return Promise.resolve(partnersgroup);
      }
    }
    return Promise.resolve(this.defaultData());
  }

  async getBusinnessPartnerCategories(): Promise<Categorie[]> {
    if (this.db.issqlite) {
      const partnerscategories = await this.db.getPartnerCategories();
      if (
        partnerscategories !== undefined &&
        partnerscategories !== null &&
        partnerscategories.length > 0
      ) {
        return Promise.resolve(partnerscategories);
      }
    }

    return Promise.resolve(this.defaultData());
  }

  defaultData(): any[] {
    const data: any[] = [];
    return data;
  }
}
