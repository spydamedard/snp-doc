import { Settings } from './../../models/user';
import { Injectable } from '@angular/core';
import { Checker} from '../../models/ui-model';
import { DatabaseService } from '../database/database.service';


@Injectable({
  providedIn: 'root',
})
export class DefaultService {
  constructor(private db: DatabaseService) {}

  async getDefaultSettings() {
    const settings = await this.db.getSettings();
    if (settings.length > 0 && settings !== undefined && settings !== null) {
      return settings[0];
    } else {
      const sett = new Settings();
      return sett;
    }
  }
  /******************************************************************************************                                                          *
   * initSelectWareHouse: this function is called by the UI for selecting the warehouse     *
   * @params:                                                                               *
   *        @param1: warehouseid: the warehouse that is going to be selected                *
   * @returned: <boolean>                                                                   *
   ****************************************************************************************/

  async initSelectWareHouse(warehid: number) {
    // we check the warehouse id
    if (warehid < 0) {
      throw 'please provide a good warehouse';
    }
    const res = await this.db.UpdateDefaultUserSetting(-1, warehid, -1);
    if (res) {
      // update default checker settings by changing the warehouse value
      const org_and_warehouse_checker = new Checker();
      org_and_warehouse_checker.id = 123;
      org_and_warehouse_checker.org = -1;
      org_and_warehouse_checker.warehouse = 2;
      await this.db.UpdateOrg_WareHouse_Checker(org_and_warehouse_checker);
      return true;
    } else {
      return false;
    }
  }

  /******************************************************************************************                                                          *
   * initSelectRole: this function is called by the UI for selecting the role               *
   * @params:                                                                               *
   *        @param1: roleid: the roleid that is going to be selected                        *
   * @returned: <boolean>                                                                   *
   ****************************************************************************************/

  async initSelectRole(roleid: number) {
    // we check the roleid
    if (roleid <= 0) {
      throw 'please provide a good role';
    }
    // added await (Medard)
    const r_query = await this.db.getRole(roleid);
    // added additional check for length (Medard)
    if (r_query !== null && r_query !== undefined && r_query.length > 0) {
      const res = await this.db.UpdateDefaultUserSetting(-1, -1, roleid);
      if (res) {
        return true;
      } else {
        return false;
      }
    } else {
      throw 'the role does not exist';
    }
  }

  /******************************************************************************************                                                          *
   * initConnexionAftSelection: this function is called by the UI for ending  the login     *
   *                   @params:                                                             *
   *                      empty                                                             *
   *      @returned: <boolean>                                                                *
   ****************************************************************************************/
  async initConnexionAftSelection() {
    const checker = await this.db.getOrgWarehouseChecker(123);
    if (checker !== undefined && checker !== null) {
      if (checker[0].ORG === 2 && checker[0].WAREHOUSE === 1) {
        throw 'Please select the warehouse';
      } else if (checker[0].ORG === 1 && checker[0].WAREHOUSE === 2) {
        throw 'Please select the organization ';
      } else if (checker[0].ORG === 1 && checker[0].WAREHOUSE === 1) {
        throw 'Please select the organization and the warehouse ';
      } else {
        return true;
      }
    } else {
      return false;
    }
  }
}
