import { OrderService } from './../order/order.service';
import { DefaultService } from '../default/default.service';
import { DatabaseService } from '../database/database.service';
import { Injectable } from '@angular/core';
import { UOMConversion } from '../../models/ui-model';
import { MProduct } from '../../models/product-data';

@Injectable({
  providedIn: 'root',
})
export class UOMService {
  constructor(
    private db: DatabaseService,
    private order: OrderService,
    private defaultserv: DefaultService
  ) {}

  /* ****************************************************************************
    updateUOM: this function updates the order line product unity of mesure (UOM)
    @params
        orderid: the current order
      productid: the product that we want to change the unity of mesure

    @return
          Promise <boolean>
  *******************************************************************************/
  async updateUOM(orderid: any, productId: any) {
    this.checkParametersBeforeInit(orderid,productId);
    // initialization of UOM
    let uomidForChangingOrderLineUnitOfMeasure = 0;
    // initialization of his devideRate
    let uomDivideRateForChangingOrderLineUnitOfMeasure = 0;
    //here we get the current order line by using the order id and the product id
    const lineForChangingOrderLineUnitOfMeasure = await this.order.getOrderLine(orderid, '', productId);
    //unit of measure conversions
    const uomConversionsForChangingOrderLineUnitOfMeasure = await this.getUnitOfMeasureConversions(productId);
    // initialization of differents vaiables
    let firstItemForChangingOrderLineUnitOfMeasure = 0;
    let kForChangingOrderLineUnitOfMeasure = 0;
    let seeForChangingOrderLineUnitOfMeasure = false;
    while (kForChangingOrderLineUnitOfMeasure <= uomConversionsForChangingOrderLineUnitOfMeasure.length && !seeForChangingOrderLineUnitOfMeasure) {
      if (
        uomConversionsForChangingOrderLineUnitOfMeasure[kForChangingOrderLineUnitOfMeasure].C_UOM_To_ID === lineForChangingOrderLineUnitOfMeasure[0].C_UOM_ID &&
        uomConversionsForChangingOrderLineUnitOfMeasure[kForChangingOrderLineUnitOfMeasure + 1] !== undefined
      ) {
        uomidForChangingOrderLineUnitOfMeasure = uomConversionsForChangingOrderLineUnitOfMeasure[kForChangingOrderLineUnitOfMeasure + 1].C_UOM_To_ID;
        seeForChangingOrderLineUnitOfMeasure = true;
        uomDivideRateForChangingOrderLineUnitOfMeasure = uomConversionsForChangingOrderLineUnitOfMeasure[kForChangingOrderLineUnitOfMeasure + 1].DivideRate;
      } else if (uomConversionsForChangingOrderLineUnitOfMeasure[kForChangingOrderLineUnitOfMeasure + 1] === undefined) {
        uomidForChangingOrderLineUnitOfMeasure = firstItemForChangingOrderLineUnitOfMeasure;
        seeForChangingOrderLineUnitOfMeasure = true;
      }
      // we save the first UOM conversion item
      if (kForChangingOrderLineUnitOfMeasure === 0) {
        firstItemForChangingOrderLineUnitOfMeasure = uomConversionsForChangingOrderLineUnitOfMeasure[kForChangingOrderLineUnitOfMeasure].C_UOM_To_ID;
      }
      kForChangingOrderLineUnitOfMeasure++;
    }

    if ((lineForChangingOrderLineUnitOfMeasure === undefined && lineForChangingOrderLineUnitOfMeasure === null) || lineForChangingOrderLineUnitOfMeasure.length === 0) {
      throw new Error('The order line is not defined');
    }
    // get settings
    const settingForChangingOrderLineUnitOfMeasure = await this.defaultserv.getDefaultSettings();
    // default price list
    const pricelistidForChangingOrderLineUnitOfMeasure = settingForChangingOrderLineUnitOfMeasure.DefaultPriceList;
    const verForChangingOrderLineUnitOfMeasure = await this.db.getPriceListVersion(pricelistidForChangingOrderLineUnitOfMeasure, 1);
    // get product price based on price list version and product id
    const ppriceForChangingOrderLineUnitOfMeasure = await this.db.getProductPrice(
      productId,
      verForChangingOrderLineUnitOfMeasure[0].M_PriceList_Version_ID
    );
    // here we recieve the current order line quatity entered and the actual price
    const qtyForChangingOrderLineUnitOfMeasure = lineForChangingOrderLineUnitOfMeasure[0].QtyEntered;
    const PricestdForChangingOrderLineUnitOfMeasure = ppriceForChangingOrderLineUnitOfMeasure[0].PriceStd;
    const PriceActualForChangingOrderLineUnitOfMeasure = lineForChangingOrderLineUnitOfMeasure[0].PriceActual;
    const current_net_priceForChangingOrderLineUnitOfMeasure = lineForChangingOrderLineUnitOfMeasure[0].LineNetAmt;
    const order_line_idForChangingOrderLineUnitOfMeasure = lineForChangingOrderLineUnitOfMeasure[0].C_OrderLine_ID;
    // here we recieve the dividerate
    let price = PriceActualForChangingOrderLineUnitOfMeasure;
    let netprice = 0;
    price=(uomDivideRateForChangingOrderLineUnitOfMeasure === 0)?PricestdForChangingOrderLineUnitOfMeasure:PricestdForChangingOrderLineUnitOfMeasure * uomDivideRateForChangingOrderLineUnitOfMeasure;
    // here we set the net price
    netprice = price * qtyForChangingOrderLineUnitOfMeasure;
    //here we are going to calculate the new grand total
    const order = await this.db.getOrder(orderid, '', '', '', '');
    const current_grand_total = order[0].GrandTotal;
    // first all we will remove the current product value
    const totalMinusCurrentProductValue =
      current_grand_total - current_net_priceForChangingOrderLineUnitOfMeasure;
    // now we calculate the new grand total line[0].C_OrderLine_ID
    const new_grand_total = totalMinusCurrentProductValue + netprice;
    return this.updateOrderLineElements(
      orderid,
      order_line_idForChangingOrderLineUnitOfMeasure,
      uomidForChangingOrderLineUnitOfMeasure,
      netprice,
      price,
      new_grand_total
    );
  }

  async getUnitOfMeasureConversions(productIdForFetchingUnitOfMeasureConversion: any) {
    // get product
    let uomConversionsForFetchingUnitOfMeasureConversion: UOMConversion[] = [];
    const productForFetchingUnitOfMeasureConversion: MProduct[] = await this.db.getProducts(productIdForFetchingUnitOfMeasureConversion, 0);
    if (productForFetchingUnitOfMeasureConversion === undefined || productForFetchingUnitOfMeasureConversion === null || productForFetchingUnitOfMeasureConversion.length === 0) {
      throw new Error('product does not exist');
    }
    let line_uomForFetchingUnitOfMeasureConversion = new UOMConversion();
    line_uomForFetchingUnitOfMeasureConversion.C_UOM_ID = productForFetchingUnitOfMeasureConversion[0].C_UOM_ID;
    line_uomForFetchingUnitOfMeasureConversion.C_UOM_To_ID = productForFetchingUnitOfMeasureConversion[0].C_UOM_ID;
    line_uomForFetchingUnitOfMeasureConversion.DivideRate = 1;
    line_uomForFetchingUnitOfMeasureConversion.Name = 'base';
    uomConversionsForFetchingUnitOfMeasureConversion.push(line_uomForFetchingUnitOfMeasureConversion);
    //here we get all the UOM product
    const allUOMCONversionForFetchingUnitOfMeasureConversion = await this.db.getProductUOM(productIdForFetchingUnitOfMeasureConversion, 0, 0);

    if (
      allUOMCONversionForFetchingUnitOfMeasureConversion === undefined ||
      allUOMCONversionForFetchingUnitOfMeasureConversion === null ||
      allUOMCONversionForFetchingUnitOfMeasureConversion.length === 0
    ) {
      throw new Error('cannot fetch products Unit of measure');
    }

    for (const conv of allUOMCONversionForFetchingUnitOfMeasureConversion) {
      line_uomForFetchingUnitOfMeasureConversion = new UOMConversion();
      line_uomForFetchingUnitOfMeasureConversion.C_UOM_ID = conv.C_UOM_ID;
      line_uomForFetchingUnitOfMeasureConversion.C_UOM_To_ID = conv.C_UOM_To_ID;
      line_uomForFetchingUnitOfMeasureConversion.DivideRate = conv.DivideRate;
      line_uomForFetchingUnitOfMeasureConversion.Name = conv.Name;
      uomConversionsForFetchingUnitOfMeasureConversion.push(line_uomForFetchingUnitOfMeasureConversion);
    }

    return uomConversionsForFetchingUnitOfMeasureConversion;
  }

  checkParametersBeforeInit(orderidChecker: any, productIdChecker: any) {
    //first of all we check the enters parameters
    if (orderidChecker === null || orderidChecker === undefined) {
      throw new Error('please select the order');
    }
    if (productIdChecker === null || productIdChecker === undefined) {
      throw new Error('you should provide the product');
    }
  }

  async updateOrderLineElements(
    orderidParameterForUpdatingOrderline: any,
    order_line_idParameterForUpdatingOrderline: string,
    uomidParameterForUpdatingOrderline: number,
    netpriceParameterForUpdatingOrderline: number,
    priceParameterForUpdatingOrderline: number,
    new_grand_totalParameterForUpdatingOrderline: number
  ) {
    // now we are going to mke updates
    // update line UOM
    const resVariableForUpdatingOrderline = await this.db.updateOrderLine(
      order_line_idParameterForUpdatingOrderline,
      'uom',
      uomidParameterForUpdatingOrderline.toString()
    );

    if (resVariableForUpdatingOrderline.rowsAffected !== 1) {
      throw new Error('could not update order line item UOM');
    }
    // update line netamount
    const res2VariableForUpdatingOrderline = await this.db.updateOrderLine(
      order_line_idParameterForUpdatingOrderline,
      'netamount',
      netpriceParameterForUpdatingOrderline.toString()
    );
    if (res2VariableForUpdatingOrderline.rowsAffected !== 1) {
      throw new Error('could not update order line item net amount');
    }
    // update line netamount
    const res3VariableForUpdatingOrderline = await this.db.updateOrderLine(
      order_line_idParameterForUpdatingOrderline,
      'actualprice',
      priceParameterForUpdatingOrderline.toString()
    );

    if (res3VariableForUpdatingOrderline.rowsAffected !== 1) {
      throw new Error('could not update order line item price actual');
    }
    // update order grand total with new total
    await this.order.updateOrder(
      orderidParameterForUpdatingOrderline,
      '',
      new_grand_totalParameterForUpdatingOrderline,
      0
    );
    return this.order.getOLine(
      orderidParameterForUpdatingOrderline,
      order_line_idParameterForUpdatingOrderline
    );
  }

  async returnProductUOM(productid: number) {
    // get product
    const product: MProduct[] = await this.db.getProducts(productid, 0);
    if (product === undefined || product === null || product.length === 0) {
      throw new Error('product does not exist');
    }

    return product.length;
  }
}
