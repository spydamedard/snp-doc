import { DatabaseService } from '../database/database.service';
import { Injectable } from '@angular/core';
import { Log } from '../../models/log';

@Injectable({
  providedIn: 'root',
})
export class LogService {
  constructor(private databaseServe: DatabaseService) {}

  async newLog(log: Log): Promise<boolean> {

    // we are checking the data key
    if (log.L_key === '' || log.L_key === undefined) {
      throw new Error('Create log failed. the key must be provided');
    }

    if (log.L_created === '' || log.L_created === undefined) {
      throw new Error('Create log failed. the created date must be provided');
    }

    if (log.L_msg === '' || log.L_msg === undefined) {
      throw new Error('Create log failed. the message cannot be empty');
    }

    if (log.L_id === '' || log.L_id === undefined) {
      throw new Error('Create log failed. the log id must be provided');
    }

    const logStrategy = await this.databaseServe.getLogStrategy();

    if (
      logStrategy !== undefined &&
      logStrategy !== null &&
      logStrategy.length > 0
    ) {
      if (logStrategy[0].IsEnabled === 'N') {
        return false;
      }
    } else {
      return false;
    }

    let result = false;
    const res = await this.databaseServe.SaveNewLog(log).catch((error) => {
      this.handleError(error);
    });

    if (res.rowsAffected === 1) {
      result = true;
    }
    return result;
  }

  async displayLogs(
    key = '',
    value = '',
    page = 0,
    size = 100
  ): Promise<Log[]> {
    const logStrategy = await this.databaseServe.getLogStrategy();
    if (
      logStrategy !== undefined &&
      logStrategy !== null &&
      logStrategy.length > 0
    ) {
      if (logStrategy[0].IsEnabled === 'N') {
        throw new Error('log strategy is disabled');
      }
    } else {
      throw new Error('unable to find the log strategy');
    }
    return this.databaseServe.getLogs(key, value, page, size);
  }

  async deleteLog(key = '', value = ''): Promise<boolean> {
    let result = false;
    const res = await this.databaseServe.DeleteLog(key, value);
    if (res !== null && res !== undefined) {
      result = true;
    }

    return result;
  }

  async changeLogStrategy(logStrag: string) {
    if (logStrag === '') {
      throw new Error('The strategy cannot be empty');
    }

    if (logStrag === undefined) {
      throw new Error('The log strategy is undefined');
    }

    let result = false;
    const lstrag = await this.databaseServe.createLogStartegy(logStrag);

    if (lstrag !== null) {
      result = true;
    }
    return result;
  }

  async getLogStrategy() {
    return this.databaseServe.getLogStrategy();
  }

  generateLogId(): string {
    return (
      '' +
      new Date().getFullYear() +
      new Date().getMonth() +
      new Date().getDate() +
      new Date().getHours() +
      new Date().getMinutes() +
      new Date().getSeconds() +
      Math.floor(Math.random() * (100 + 1))
    );
  }

  private handleError(error: any) {
    throw error;
  }
}
